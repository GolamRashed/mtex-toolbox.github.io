plot

(method of quaternion)

Syntax

varargout = plot(q,'param1',val1,'param2',val2)

Example 

x = -pi:pi/10:pi;
y = tan(sin(x)) - sin(tan(x));
plot(x,y,'--rs','LineWidth',2,...
'MarkerEdgeColor','k',...
'MarkerFaceColor','g',...
'MarkerSize',10)
